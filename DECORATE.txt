ACTOR Fist : Weapon
{
  Game Doom
  Weapon.SelectionOrder 3700
  weapon.kickback 0
  Obituary "$OB_MPFIST"
  Tag "$TAG_FIST"
  +INVENTORY.UNDROPPABLE
  +WEAPON.MELEEWEAPON
  +WEAPON.WIMPY_WEAPON
  States
  {
  Ready:
    PUNG A 1 A_WeaponReady
    Loop
  Deselect:
    PUNG A 0 A_Lower
	PUNG A 0 A_Lower
	PUNG A 0 A_Lower
	PUNG A 1 A_Lower
    Loop
  Select:
    PUNG A 0 A_Raise
	PUNG A 0 A_Raise
	PUNG A 0 A_Raise
	PUNG A 1 A_Raise
    Loop
  Fire:
    PUNG B 2
    TNT1 A 0 A_JumpIfInventory("PowerStrength", 1, "Berserked")
  Normal:
    PUNG C 4 A_CustomPunch(50,FALSE)
    Goto FireEnd
  Berserked:
    PUNG C 4 A_CustomPunch(200,FALSE)
  FireEnd:
    PUNG D 3
    PUNG C 2
    PUNG B 3 A_ReFire
    Goto Ready
  }
}
